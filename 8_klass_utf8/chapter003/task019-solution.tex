  \begin{gathered}
    \left[
      \begin{gathered}
        2x+9=x, \hfill
        \\
        2x+9=-x. \hfill
        \\
      \end{gathered}
    \right. \hfill
    \\
    x\geqslant 0.    \\
  \end{gathered}
\end{cases}\Leftrightarrow \begin{cases}
  \begin{gathered}
    \left[
      \begin{gathered}
        x=-9, \hfill
        \\
        x=-3. \hfill
        \\
      \end{gathered}
    \right. \hfill
    \\
    x\geqslant 0.    \\
  \end{gathered}
\end{cases}
\Leftrightarrow x \in \{\varnothing\}.$\\
